168  notEmpty = spiRecvQueue[spiRecvQueueIdx][i] != 0 ? true : notEmpty;

171  if (notEmpty) {

173  spiRecvQueue[spiRecvQueueIdx][0] = 0xaa;

176  spiRecvQueueIdx = 0;

182  notEmpty = spiRecvQueue[spiRecvQueueIdx][i] != 0 ? true : notEmpty;
183  // if there is data, receive buf is too small
